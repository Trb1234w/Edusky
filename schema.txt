-- closure maintenance
create or replace function public.categories_closure_after_insert() returns trigger as $$
begin
  insert into public.categories_closure (ancestor, descendant, depth) values (new.id, new.id, 0)
    on conflict do nothing;

  if new.parent_id is not null then
    insert into public.categories_closure (ancestor, descendant, depth)
    select ancestor, new.id, depth + 1
    from public.categories_closure
    where descendant = new.parent_id
    on conflict do nothing;
  end if;

  return new;
end;
$$ language plpgsql;

drop trigger if exists trg_categories_after_insert on public.categories;
create trigger trg_categories_after_insert
  after insert on public.categories
  for each row execute procedure public.categories_closure_after_insert();

-- update parent handler (simplifié & safe)
create or replace function public.categories_closure_after_update() returns trigger as $$
begin
  -- For simplicity: on parent change, delete all closure rows where descendant in subtree and ancestor not valid,
  -- then reinsert closure for new parent. A robust approach may require full rebuild; keep this simple.
  if old.parent_id is distinct from new.parent_id then
    -- remove relations where ancestor is not equal to self and related to old ancestry
    delete from public.categories_closure
    where descendant in (select descendant from public.categories_closure where ancestor = old.id)
      and ancestor in (select ancestor from public.categories_closure where descendant = old.id and ancestor <> old.id);

    -- reinsert closure relationships for new parent
    insert into public.categories_closure (ancestor, descendant, depth)
    select ancestor, d.descendant, ancestor_depth + d.depth + 1
    from public.categories_closure a
    join (select descendant as descendant, depth from public.categories_closure where ancestor = new.id) d on true
    left join lateral (
      select ancestor as ancestor, depth as ancestor_depth from public.categories_closure where descendant = new.parent_id
    ) anc on true
    on conflict do nothing;

    -- ensure self relation
    insert into public.categories_closure (ancestor, descendant, depth) values (new.id, new.id, 0) on conflict do nothing;
  end if;
  return new;
end;
$$ language plpgsql;

drop trigger if exists trg_categories_after_update on public.categories;
create trigger trg_categories_after_update
  after update on public.categories
  for each row execute procedure public.categories_closure_after_update();

create table if not exists public.avis_formation (

  formation_id uuid references public.formations(id) on delete cascade,
  user_id uuid references public.profiles(id) on delete set null,

create index if not exists idx_avis_formation_formation on public.avis_formation (formation_id);
create index if not exists idx_avis_formation_user on public.avis_formation (user_id);

create or replace function public.recompute_formations_ratings() returns trigger as $$
declare
  avgnote numeric;
  cnt int;
  fid uuid;
begin
  if (TG_OP = 'DELETE') then
    fid := OLD.formation_id;
  else
    fid := NEW.formation_id;
  end if;

  select coalesce(avg(note),0), count(*) into avgnote, cnt from public.avis_formation where formation_id = fid;

  update public.formations set note_moyenne = round(avgnote::numeric,2), nb_avis = cnt where id = fid;

  return null;
end;
$$ language plpgsql;

drop trigger if exists trg_avis_after_insert on public.avis_formation;
create trigger trg_avis_after_insert
  after insert on public.avis_formation
  for each row execute procedure public.recompute_formations_ratings();

drop trigger if exists trg_avis_after_update on public.avis_formation;
create trigger trg_avis_after_update
  after update on public.avis_formation
  for each row execute procedure public.recompute_formations_ratings();

drop trigger if exists trg_avis_after_delete on public.avis_formation;
create trigger trg_avis_after_delete
  after delete on public.avis_formation

create or replace view public.vue_professeurs_annuaire as
select
  prof.id,
  coalesce(pr.prenom || ' ' || pr.nom, pr.full_name) as nom_complet,
  pr.pays_id,
  pr.ville_id,
  pr.quartier_id,
  prof.titre,
  prof.presentation,
  prof.specialites,
  prof.note_moyenne,
  prof.is_publie,
  pr.avatar_url,
  prof.nb_etudiants_formes
from public.professeurs prof
join public.profiles pr on pr.id = prof.id
where prof.is_publie = true;

create or replace view public.vue_formations_public as
select
  f.*,
  pr.full_name as professeur_nom
from public.formations f
left join public.professeurs pf on pf.id = f.professeur_id
left join public.profiles pr on pr.id = pf.id
where f.statut = 'publie';
